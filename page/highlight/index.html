使用方法

        
            <!-- 引入主题文件 -->
            <link rel="stylesheet" href="path/${theme}.css">
            <!-- 引入js文件 -->
            <script src="/path/highlight.js"></script>
        
        
            document.addEventListener('DOMContentLoaded', (event) => {
                document.querySelectorAll('pre code').forEach((block) => {
                    hljs.highlightBlock(block);
                });
            });
